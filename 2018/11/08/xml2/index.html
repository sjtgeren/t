package com.briup.sax;

import java.io.File;

import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.transform.Transformer;
import javax.xml.transform.TransformerFactory;
import javax.xml.transform.dom.DOMSource;
import javax.xml.transform.stream.StreamResult;

import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.Text;

public class DOMParser{
//创建xml文件
public void creatXML()throws Exception{
DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
DocumentBuilder builder = factory.newDocumentBuilder();
//生成document
Document document = builder.newDocument();
Element books = document.createElement(“books”);
document.appendChild(books);
Element book = document.createElement(“book”);
books.appendChild(book);
//创建元素
Element name = document.createElement(“name”);
name.setAttribute(“id”,”1”);//添加属性
Element price = document.createElement(“price”);
//设置文本内容
Text nameText = document.createTextNode(“coreJava”);
Text priceText = document.createTextNode(“100”);
name.appendChild(nameText);
price.appendChild(priceText);
book.appendChild(name);
book.appendChild(price);
TransformerFactory factory2 = TransformerFactory.newInstance();
Transformer transformer = factory2.newTransformer();
transformer.transform(new DOMSource(document),
new StreamResult(
new File(“src/com/briup/sax/books.xml”)));
System.out.println(“文件创建成功”);
}
public static void main(String[] args) {
try {
new DOMParser().creatXML();
} catch (Exception e) {
// TODO Auto-generated catch block
e.printStackTrace();
}
}
}

package com.briup.sax;

import java.io.File;
import java.util.ArrayList;
import java.util.List;

import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;

import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.NodeList;

public class StudentDom {
public static void main(String[] args) throws Exception {
Student stu = null;
List list = new ArrayList();
// 1.创建工厂类对象
DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
// 创建解析对象
DocumentBuilder builder = factory.newDocumentBuilder();
// 调用解析方法
File f = new File(“src/com/briup/sax/stus.xml”);
// document代表的就是xml文件，文档树的根
Document document = builder.parse(f);
NodeList nodeList = document.getElementsByTagName(“stu”);
for (int i = 0; i < nodeList.getLength(); i++) {
// 获取集合中具体某一个节点
Element node = (Element) nodeList.item(i);
stu = new Student();
int id = Integer.parseInt(node.getAttribute(“id”));
int money = Integer.parseInt(node.getAttribute(“money”));
stu.setId(id);
stu.setMoney(money);
// 获取当前节点中子节点
NodeList childList = node.getChildNodes();
for (int j = 0; j < childList.getLength(); j++) {
// 获取当前节点的类型，去除文本节点(内容)
if (childList.item(j).getNodeType() == 1) {
// 获取元素名
String nodeName = childList.item(j).getNodeName();
// 获取对应元素的文本内容
String nodeValue = childList.item(j).getFirstChild().getNodeValue();
switch (nodeName) {
case “name”:
stu.setName(nodeValue);
break;
case “age”:
stu.setAge(Integer.parseInt(nodeValue));
break;
case “gender”:
stu.setGender(nodeValue);
break;
case “like”:
stu.setLike(nodeValue);

break;

default:
break;
}
}
}
list.add(stu);
}
for (Student student : list) {

System.out.println(student);
}
}
}





package com.briup.sax;

import java.io.File;
import java.util.ArrayList;
import java.util.List;

import javax.xml.parsers.SAXParser;
import javax.xml.parsers.SAXParserFactory;

import org.xml.sax.Attributes;
import org.xml.sax.SAXException;
import org.xml.sax.helpers.DefaultHandler;

public class StudentSAX {
public static void main(String[] args) throws Exception{
//1.sax解析的工厂类对象
SAXParserFactory factory = SAXParserFactory.newInstance();
//解析类，将数据包装到类中，之后放到集合中
//2.创建SAX解析器
SAXParser parser = factory.newSAXParser();
//3.解析xml
File f=new File(“src/com/briup/sax/stus.xml”);
StudentHandler dh=new StudentHandler();
parser.parse(f, dh);

}
}
class StudentHandler extends DefaultHandler{
private List list;
private Student stu;
private String tagName;
//文档开始解析的通知
@Override
public void startDocument() throws SAXException {
list=new ArrayList();
}
//读取开始元素
//uri:文件路径
//localName:命名空间
//qName:开始标签名
//attributes:开始标签的属性
@Override
public void startElement(String uri, String localName, String qName,
Attributes attributes) throws SAXException {
if(“stu”.equals(qName)){
stu=new Student();
//获取属性
int id=Integer.parseInt(attributes.getValue(“id”));
int money=Integer.parseInt(attributes.getValue(“money”));
stu.setId(id);
stu.setMoney(money);
}else{
tagName=qName;
}
}

//获取元素内容
@Override
public void characters(char[] ch, int start, int length)
throws SAXException {
//只是获取文本内容，没有标签名
String str=new String(ch,start,length);
if(“name”.equals(tagName)){
stu.setName(str);
}else if(“age”.equals(tagName)){
stu.setAge(Integer.parseInt(str));
}else if(“gender”.equals(tagName)){
stu.setGender(str);
}else if(“like”.equals(tagName)){
stu.setLike(str);
}

​
}
//读取结束元素
//uri:文件路径
//localName:命名空间
//qName:结束标签名
@Override
public void endElement(String uri, String localName, String qName)
throws SAXException {
if(“stu”.equals(qName)){
list.add(stu);
}
tagName=null;//为了把换行符去掉
}
@Override
public void endDocument() throws SAXException {
System.out.println(list.size());
for (Student stu : list) {
System.out.println(stu);
}
}
}
