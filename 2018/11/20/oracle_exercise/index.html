oracle第六天笔记

发表于 2018-11-20 | 更新于: 2018-11-20 | 0 comments | 热度: ℃ | 阅读数
字数统计: 1,762字 | 阅读时长 ≈ 9分钟

1)查询所有员工的平均工资，最高工资最低工资，工资总和，
还有有多少个员工?
select avg(salary),max(salary),min(salary),
sum(salary),count(id) from s_emp;
2)查询每个部门的平均工资？对平均工资降序排序.
平均工资大于1400.
select avg(salary) from s_emp group by dept_id
having avg(salary)>1400 order by avg(salary) desc;
3）查看各个部门的最高工资
Select dept_id,max(salary)
From s_emp
Group by dept_id;

4)查看各个部门的员工数
Select dept_id,count() 人数
From s_emp
Group by dept_id;

5)查询各个部门各个职称的平均薪水和最大薪水,
并且平均薪水大于2000的部门id。
Select dept_id,title,avg(salary) avg,max(salary) max
From s_emp
Group by dept_id,title having avg(salary)>2000;

6)查询title中不包含vp字符串的每个职位的平
均薪水，并对平均薪水进行降序排列，并且每个职位
的总薪水大于5000。
select avg(salary),title from s_emp where lower(title)
not like ‘%vp%’ group by title
having sum(salary)>5000 order by avg(salary) desc;
7)查看每个区域顾客的人数？
Select region_id,count()人数
From s_customer
Group by region_id;

8)查看每个区域部门数？
Select region_id,count()人数
From s_dept
Group by region_id;

9)查询南美地区的部门数？
Select count(),s_region.name
From s_dept,s_region
Where region_id=s_region.id and
lower(s_region.name)=’south america’
group by s_region.name;

10)查询南美地区工资大于1400的员工的信息？
select e.last_name,d.name,r.name from s_emp e,
s_dept d,s_region r where e.salary>1400 and
e.dept_id=d.id and d.region_id=r.id and
lower(r.name)=’south america’;





1)查询所有员工的ID，名字和所在部门的名称
column first_name format a20
Column name format a20
Select s_emp.id,first_name,name
From s_emp,s_dept
Where dept_id=s_dept.id;

2)查询员工的姓名和部门所在地区的名称
Select s_emp.id,s_region.name
From s_emp,s_dept,s_region
Where dept_id=s_dept.id and region_id=s_region.id;

3）查询部门名称包含Sa的员工姓名薪水
Select first_name,salary,name From s_emp,s_dept Where upper(name) like’%SA%’ and dept_id=s_dept.id;

4）查询欧洲销售部门的薪水在1000到2000的员工信息
Select e.last_name,e.salary,r.name,d.name
From s_emp e,s_dept d,s_region r
Where e.dept_id=d.id and d.region_id=r.id
And lower(r.name) =’europe’
and lower(d.name) =’sales’
And e.salary between 1000 and 2000;

5）查询部门名称是5位，该部门员工的薪水不等于1500，并按员工的薪水降序排序
Select first_name,salary,name
From s_emp,s_dept
Where dept_id=s_dept.id
And length(name)=5
And salary !=1500
Order by salary desc;

6）创建下面这张表并插入数据
create table s_gender(
id number(5) primary key,
minSal number(7),
maxSal number(7),
name varchar2(20));
insert into s_gender values(1,0,1000,’蓝领’);
insert into s_gender values(2,1000,1500,’白领’);
insert into s_gender values(3,1500,2500,’金领’);
commit;
然后查询所有员工月薪涨500后的工资等级？
Select first_name,salary+500,s_gender.name
From s_emp,s_gender
Where salary+500 between minSal and maxSal;

7）查询所有员工所在部门的名称，但是
需要把所有部门给查询出来.
insert into s_dept values(999,’教学部’,2);
commit;
方案一：
Select s_emp.id,first_name,name
From s_emp,s_dept
Where dept_id(+)=s_dept.id;
方案二：
Select s_emp.id,first_name,name
From s_emp right join s_dept on dept_id=s_dept.id;

8）查询员工所在部门的名称，包括没有部门号的员工
方案一：
Select s_emp.id,first_name,dept_id,name
From s_emp,s_dept
Where dept_id=s_dept.id(+);
方案二：
Select s_emp.id,first_name,dept_id,name
From s_emp left join s_dept on dept_id=s_dept.id;
9）查询员工所在部门的信息，
没有部门号的员工
所有的部门都查询出来
方案一：
Select s_emp.id,first_name,s_dept.id,name,region_id
From s_emp,s_dept
Where dept_id=s_dept.id(+)
union
Select s_emp.id,first_name,s_dept.id,name,region_id
From s_emp,s_dept
Where dept_id(+)=s_dept.id;
方案二：
Select s_emp.id,first_name,name
From s_emp full join s_dept on dept_id=s_dept.id;



10）查询所有员工的经理的last_name,salary包括没有经理的员工。
Column last_name format a20
select w.last_name,m.last_name,m.salary
from s_emp w,s_emp m
where w.manager_id=m.id(+);

11）部门表中前7条记录？
Select id,name,region_id
From s_dept
Where rownum<=7;
12）部门表中第3条到第5条数据？
Select from s_dept
Where rownum<=5
Minus
Select from s_dept
Where rownum<3;

1）查看和Ngao所在部门的区域id一样的所有员工id和名字
Column last_name format a20
步骤一：
Select region_id
From s_emp,s_dept
Where dept_id=s_dept.id and last_name=’Ngao’;
步骤二：
Select s_emp.id,last_name
From s_emp,s_dept
Where dept_id=s_dept.id and region_id=(
Select region_id
From s_emp,s_dept
Where dept_id=s_dept.id and last_name=’Ngao’
);
2）查询每一个部门的员工小于该部门的平均工资的员工的信息？
步骤一：
Select avg(salary) avg,dept_id
from s_emp
Group by dept_id;
方案一步骤二：
Select last_name,salary,s_emp.dept_id ,avg
From s_emp,(
Select avg(salary) avg,dept_id
From s_emp
Group by dept_id
) a
Where s_emp.dept_id=a.dept_id and salary<avg;

//一次性写
方案二步骤二：
Select last_name,salary,dept_id
From s_emp e
Where salary<(
Select avg(salary)
From s_emp
Where e.dept_id=s_emp.dept_id
);
3)查看职位和名字为Chang的员工一样的所有员工id和名字
步骤一：
Select title from s_emp where last_name=’Chang’;
步骤二：
Select id,last_name,title
From s_emp
Where title=(
Select title from s_emp where last_name=’Chang’
);
4)查看员工工资小于平均工资的所有员工的id和名字
步骤一：
Select avg(salary) from s_emp;
步骤二：
Select id,last_name,salary
From s_emp
Where salary<(
Select avg(salary) from s_emp
);
5)查看部门id和名字为Chang的部门相同或者区域ID为2的部门信息
步骤一：
Select dept_id from s_emp where last_name=’Chang’;
步骤二：
Select *
From s_dept
Where id=(Select dept_id from s_emp where last_name=’Chang’) or region_id=2;
6)查询员工所在区域为2的部门的员工的信息？
步骤一：
Select id from s_dept where region_id=2;
步骤二：
Select id,last_name,dept_id
From s_emp
Where dept_id in(
Select id from s_dept where region_id=2
);
7)查看部门平均工资大于32号部门平均工资的部门id
步骤一：
Select avg(salary) from s_emp where dept_id=32;
步骤二：
Select dept_id,avg(salary)
From s_emp
Group by dept_id having avg(salary)>(Select avg(salary) from s_emp where dept_id=32);
8)查询工资大于Smith所在部门平均工资的员工的信息

Select dept_id from s_emp where last_name=’Smith’;

Select salary from s_emp where id=(

)
