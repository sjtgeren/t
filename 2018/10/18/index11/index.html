3)System.arraycopy()
数组的长度一旦定义了，就无法修改。
所以需要arraycopy()来重新分配一个新的数组，
然后将其内容拷贝进去。
int[] a = new int[5];
int[] b = new int[5];
a=b与System.arraycopy(a,0,b,0,5)有什么区别？

a=b是将数组b的值赋值给a数组
　　　System.arraycopy(a,0,b,0,5)是将a数组的值都copy到b数组中。 　　　

比如说我将a[1]=2;请问b[1]的值是多少？

如果是a=b则，b[1]=2（b1也随着改变）
如果是System.arraycopy(a,0,b,0,5)则，b[1]=数组b[1]的初始值，不会改变

{

int[] attr={2,5,-5,98,198,23};
System.out.println(“排序前:”);
for(int i=0;i<attr.length;i++){
System.out.print(attr[i]+” “);
}
System.out.println();
//外层循环次数
for(int i=0;i<attr.length-1;i++){
for(int j=0;j<attr.length-1-i;j++){
if(attr[j]>attr[j+1]){
int temp=attr[j];//临时存放最大值
attr[j]=attr[j+1];
attr[j+1]=temp;
}
}
}
System.out.println(“排序后:”);
for(int i=0;i<attr.length;i++){
System.out.print(attr[i]+” “);
}
}
}

package com.briup.ch11;
public class Queue
{
private int[] array;//队
private int length;//队的总长度
private int size;//队的当前长度
public Queue(){
array=new int[5];
this.length=5;
}
public Queue(int length){
array=new int[length];
this.length=length;
}
//判断数组越界
public void check(){
if(size>=length){//扩容
int[] newArray=new int[10+length];
System.arraycopy(array,0,newArray,0,length);
array=newArray;
length+=10;
}
}
//入队
public void add(int num){
check();
array[size]=num;
size++;
}

//出队
public int remove(){
 int num=array[0];//删除的内容
 for(int i=0;i<size-1;i++){//代表移动几次
    array[i]=array[i+1];//往前移动数据
 }
array[size-1]=0;//将最后一个数据设置默认值
size--;
 return num;
}

//队的当前长度
public int getSize(){
return size;
}
//打印队的详细信息
public void print(){
for(int i=0;i<size;i++){
System.out.print(array[i]+” “);
}
}
}

package com.briup.ch11;
public class Selection
{
public static void main(String… args){
int[] attr={2,5,-5,198,98,23};
System.out.println(“排序前:”);
for(int i=0;i<attr.length;i++){
System.out.print(attr[i]+” “);
}
System.out.println();
for(int i=0;i<attr.length-1;i++){
int position=i;
for(int j=i+1;j<attr.length;j++){
if(attr[position]>attr[j]){
position=j;
}
}
int temp=attr[i];
attr[i]=attr[position];
attr[position]=temp;
}
System.out.println(“排序后:”);
for(int i=0;i<attr.length;i++){
System.out.print(attr[i]+” “);
}
}
}
